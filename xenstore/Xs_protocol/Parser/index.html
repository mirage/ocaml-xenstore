type state =
  1. | Unknown_operation of int32
    (*

    received an unexpected message type

    *)
  2. | Parser_failed of string
    (*

    we failed to parse a header

    *)
  3. | Need_more_data of int
    (*

    we still need 'n' bytes

    *)
  4. | Packet of t
    (*

    successfully decoded a packet

    *)
